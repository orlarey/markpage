\section{Triangle commutatif}

\begin{lstlisting}
f : A -> B
g : B -> C
h : A -> C = g . f
\end{lstlisting}

\section{Carré du produit}

\begin{lstlisting}
pi1 : P -> A
pi2 : P -> B
f   : X -> A
g   : X -> B
u   : X -> P by (f, g)

pi1 . u = f
pi2 . u = g
\end{lstlisting}


\section{Coproduit}

\begin{lstlisting}
i1 : A -> S
i2 : B -> S
f  : A -> X
g  : B -> X
v  : S -> X by (f, g)

v . i1 = f
v . i2 = g
\end{lstlisting}


\section{Carré de naturalité}

\begin{lstlisting}
Fh    : F(X) -> F(Y)
Gh    : G(X) -> G(Y)
eta_X : F(X) -> G(X)
eta_Y : F(Y) -> G(Y)

Gh . eta_X = eta_Y . Fh
\end{lstlisting}


\section{Pullback}

\begin{lstlisting}
f  : A -> C
g  : B -> C
p1 : P -> A
p2 : P -> B
h  : X -> A
k  : X -> B
u  : X -> P by (h, k)

f . p1 = g . p2
f . h  = g . k
p1 . u = h
p2 . u = k
\end{lstlisting}


\section{Égaliseur (paire parallèle)}

\begin{lstlisting}
f : A -> B
g : A -> B
e : E -> A
h : Z -> A
u : Z -> E by (h)

f . e = g . e
f . h = g . h
e . u = h
\end{lstlisting}


\section{Fonctorialité}

\begin{lstlisting}
Ff   : F(A) -> F(B)
Fg   : F(B) -> F(C)
Fgof : F(A) -> F(C) = Fg . Ff
\end{lstlisting}


\section{Objet terminal}

\begin{lstlisting}
t : A -> T by ()
\end{lstlisting}


\section{Avec modificateurs (mono / epi / iso)}

\begin{lstlisting}
direction: LR
i : A -> B (mono)
p : B -> C (epi)
q : A -> C (iso)
\end{lstlisting}


\section{Cas d'erreur — équation mal typée}

L'équation \texttt{f = g} ci-dessous est mal typée (f va de A vers B, g va de
B vers C). Le typechecker doit la rejeter avec un diagnostic positionnel.


\begin{lstlisting}
f : A -> B
g : B -> C
f = g
\end{lstlisting}
